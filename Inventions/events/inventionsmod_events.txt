invention_events.5 = {
	type = country_event
	placement = root
	title = invention_events.5.t
	desc = invention_events.5.d
	flavor = invention_events.5.f
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	event_image = {
		video = "gfx/event_pictures/middleeast_engineer_blueprint.bk2"
	}
	option = {
		name = event_completion_default
		add_modifier = {
			name = division_of_labor_modifier
		}
		ai_chance = {
			base = 1
		}
	}
	option = {
		name = event_completion_sharing
		default_option = yes
		ai_chance = {
			base = 100
		}
		trigger = {
			global_var:division_of_labor_discoveries < 40
		}
		add_modifier = {
			name = division_of_labor_modifier
		}
		if = {
			limit = {
				global_var:division_of_labor_discoveries < 1
			}
			add_modifier = {
				name = division_of_labor_first_prestige
			}
			save_scope_as = originator
			every_country = {
				trigger_event = invention_events.6
			}
		}
		if = {
			limit = {
				AND = {
					NOT = {
						global_var:division_of_labor_discoveries < 1
					}
					global_var:division_of_labor_discoveries < 5
				}
			}
			add_modifier = {
				name = division_of_labor_early_prestige
			}
		}
		if = {
			limit = {
				NOT = {
					global_var:division_of_labor_discoveries < 5
				}
			}
			add_modifier = {
				name = division_of_labor_prestige
			}
		}
		change_global_variable = {
			name = division_of_labor_discoveries
			add = 1
		}
	}
}
